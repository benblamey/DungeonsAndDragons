016    package org.fest.swing.util;
017    
018    /**
019     * Understands a tuple of size 3.
020     * @param <I> the generic type of the 1st. value in this tuple.
021     * @param <II> the generic type of the 2nd. value in this tuple.
022     * @param <III> the generic type of the 3rd. value in this tuple. 
023     *
024     * @author Alex Ruiz
025     */
026    public class Triple<I, II, III> extends Pair<I, II> {
027    
028      /** The third value in this tuple. */
029      public final III iii;
030    
031      /**
032       * Creates a new </code>{@link Triple}</code>.
033       * @param i the 1st. value in this tuple.
034       * @param ii the 2nd. value in this tuple.
035       * @param iii the 3rd. value in this tuple.
036       */
037      public Triple(I i, II ii, III iii) {
038        super(i, ii);
039        this.iii = iii;
040      }
041    
042    }
