027    /**
028     * Understands mapping characters to <code>{@link KeyStroke}</code>s for locale
029     * <code>{@link java.util.Locale#ENGLISH ENGLISH}</code>.
030     *
031     * @author Alex Ruiz
032     */
033    public class KeyStrokeMappingProvider_en implements KeyStrokeMappingProvider {
034    
035      /**
036       * Returns the mapping between characters and <code>{@link KeyStroke}</code>s for locale
037       * <code>{@link java.util.Locale#ENGLISH ENGLISH}</code>.
038       * @return the mapping between characters and <code>KeyStroke</code>s for locale <code>ENGLISH</code>.
039       */
040      public Collection<KeyStrokeMapping> keyStrokeMappings() {
041        return SingletonHolder.instance;
042      }
043    
044      // Thread-safe, lazy-loading singleton.
045      private static class SingletonHolder {
046        static List<KeyStrokeMapping> instance = createMappings();
047      }
