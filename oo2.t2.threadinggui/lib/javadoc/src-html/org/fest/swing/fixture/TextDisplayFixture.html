016    package org.fest.swing.fixture;
017    
018    import java.awt.Component;
019    import java.util.regex.Pattern;
020    
021    /**
022     * Understands state verification and property value queries of GUI components that display text.
023     *
024     * @author Alex Ruiz
025     */
026    public interface TextDisplayFixture {
027    
028      /**
029       * Returns the text of this fixture's <code>{@link Component}</code>.
030       * @return the text of the managed <code>Component</code>.
031       */
032      String text();
033    
034      /**
035       * Asserts that the text of this fixture's <code>{@link Component}</code> is equal to or matches the specified
036       * <code>String</code>.
037       * @param expected the text to match. It can be a regular expression.
038       * @return this fixture.
039       * @throws AssertionError if the text of the target component is not equal to or does not match the given one.
040       */
041      TextDisplayFixture requireText(String expected);
042    
043      /**
044       * Asserts that the text of this fixture's <code>{@link Component}</code> matches the given regular expression
045       * pattern.
046       * @param pattern the regular expression pattern to match.
047       * @return this fixture.
048       * @throws NullPointerException if the given regular expression pattern is <code>null</code>.
049       * @throws AssertionError if the text of the target component does not match the given regular expression pattern.
050       * @since 1.2
051       */
052      TextDisplayFixture requireText(Pattern pattern);
053    }
