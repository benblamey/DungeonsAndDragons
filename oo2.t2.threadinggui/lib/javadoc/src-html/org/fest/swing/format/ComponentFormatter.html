016    package org.fest.swing.format;
017    
018    import java.awt.Component;
019    
020    /**
021     * Understands a formatter that returns a <code>String</code> representation of a given <code>{@link Component}</code>.
022     *
023     * @author Alex Ruiz
024     */
025    public interface ComponentFormatter {
026    
027      /**
028       * Returns a <code>String</code> representation of the given <code>{@link Component}</code>.
029       * @param c the given <code>Component</code>.
030       * @return a <code>String</code> representation of the given <code>Component</code>.
031       */
032      String format(Component c);
033    
034      /**
035       * Returns the type of <code>{@link Component}</code> this formatter supports.
036       * @return the type of <code>Component</code> this formatter supports.
037       */
038      Class<? extends Component> targetType();
039    }
